功能实现

实现慧搜功能

首先使用IDEA导入 springboot_phoenix 项目。

Controller

在这里我们定义 search() 函数

@Controller
public class IndexController {

    @Autowired
    HbaseService service;

    /**
     * 模板跳转
     *
     * @param page
     * @return
     */
    @RequestMapping("/{page}")
    public String index(@PathVariable String page) {
        return page;
    }

    /**
     * 慧搜查询
     *
     * @param keyword
     * @param model
     * @return
     */
    @RequestMapping("/search_")
    public String search(String keyword, Model model) {
        long startTime = System.currentTimeMillis();
        List<PIR> list = service.findPIRByLikeName(keyword);
        long endTime = System.currentTimeMillis();
        long useTime = endTime - startTime;
        model.addAttribute("dataList", list);
        model.addAttribute("hit", list.size());
        model.addAttribute("keyword", keyword);
        model.addAttribute("useTime", useTime);
        return "search_result";
    }

}

Service

@Service
public class HbaseServiceImpl implements HbaseService {
    @Autowired
    HbaseMapper mapper;

    public List<PIR> findPIRByLikeName(String name) {
        List<PIR> list = mapper.findPIRByLikeName("%" + name + "%");
        return list;
    }

    public void init() {
        mapper.init();
    }
}

Mapper


import net.educoder.entity.PIR;
import org.apache.ibatis.annotations.Mapper;
import org.apache.ibatis.annotations.Select;

import java.util.List;

@Mapper
public interface HbaseMapper {

    @Select("select * from PIR2 where XM like #{name}")
    List<PIR> findPIRByLikeName(String name);
}

可选的功能
