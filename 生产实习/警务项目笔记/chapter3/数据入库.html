数据入库

Excel 数据处理

实现步骤：

  1. 读取文件，解析 Excel 文件；
  2. 获取第一列的列名；
  3. 将中文转换为拼音首字母；
  4. 创建表；

将中文转换为拼音首字母

不过第一行的数据都是中文，我们存入数据库的数据需要使用英文，所以接下来我们将中文转换为拼音，在这里使用pinyin4j工具进行转换。

使用 PinyinHelper.toHanyuPinyinStringArray 即可将一个中文转换为汉字

示例程序：


import net.sourceforge.pinyin4j.PinyinHelper;

public class Test {
    public static void main(String[] args) {
        String name = "红鲤鱼与绿鲤鱼";
        char[] charArray = name.toCharArray();
        StringBuilder pinyin = new StringBuilder();
        for(int i=0; i<charArray.length; i++){
            if(Character.toString(charArray[i]).matches("[\\u4E00-\\u9FA5]+")){  //判断是否为中文
                String value = PinyinHelper.toHanyuPinyinStringArray(charArray[i])[0];
                value = value.substring(0,value.length()-1);
                pinyin.append(value);  // 可能有多音字
            }else{
                pinyin.append(charArray[i]);
            }
        }
        System.out.println(pinyin);
    }
}

运行结果：hongliyuyulu:liyu

try {
    PreparedStatement preparedStatement = connection.prepareStatement(insertSQL.toString());
    // 批量插入数据
    for (Row cells : sheet) {
        if (cells.getRowNum() == 0){
            continue;//跳过第一行
        }
        // 获取最后一个单元格位置
        int lastCellNum = cells.getLastCellNum();

        for(int i = 1; i <= lastCellNum;i++){
            Cell cell = cells.getCell(i,Row.MissingCellPolicy.RETURN_BLANK_AS_NULL);
            if(cell!= null){
                preparedStatement.setString(i,cell.getStringCellValue());
            }else {
                preparedStatement.setString(i,""); //设置空字符串
            }
        }
        preparedStatement.addBatch();
        if (cells.getRowNum()%1000 == 0){
            preparedStatement.executeBatch(); // 在一千条的时候提交一次
        }
    }
    // 批量操作
    preparedStatement.executeBatch();
    connection.commit();
    connection.close();
} catch (SQLException e) {
    e.printStackTrace();

完整代码

import net.sourceforge.pinyin4j.PinyinHelper;
import org.apache.poi.hssf.usermodel.HSSFWorkbook;
import org.apache.poi.ss.usermodel.Cell;
import org.apache.poi.ss.usermodel.Row;
import org.apache.poi.ss.usermodel.Sheet;
import org.apache.poi.xssf.usermodel.XSSFWorkbook;

import java.io.IOException;
import java.io.InputStream;
import java.sql.*;
import java.util.ArrayList;
import java.util.List;

public class SaveCsv{
    static {
        // 1. 注册驱动
        try {
            Class.forName("org.apache.phoenix.jdbc.PhoenixDriver");
        } catch (ClassNotFoundException e) {
            e.printStackTrace();
        }
    }

    /*
     * 连接数据库
     */
    public static Connection getConnection(){

        Connection conn = null;
        try {
            //2, 获取连接对象
            conn = DriverManager.getConnection("jdbc:phoenix:116.196.80.141:2181");
        } catch (SQLException e) {
            e.printStackTrace();
        }
        return conn;
    }
    /**
     * 创建表
     */
    public static void createTable(String sql){
        Connection connection = getConnection();
        try {
            Statement statement = connection.createStatement();
            //3. 执行SQL语句
            statement.execute(sql);
            //3. 提交
            connection.commit();
            connection.close();
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }

    public static void main(String[] args) throws IOException {
        // 读取数据
        String filePath = "hotel.xlsx";
        String fileType=filePath.substring(filePath.lastIndexOf("."));//获得后缀名
        // 读一行数据
        InputStream inputStream = SaveCsv.class.getClassLoader().getResourceAsStream(filePath);
        Sheet sheet = null;
        if (fileType.equals(",xls")){
            sheet = new HSSFWorkbook(inputStream).getSheetAt(0);
        }else {
            sheet = new XSSFWorkbook(inputStream).getSheetAt(0);
        }
        // 获取行数
        int rowNum=sheet.getLastRowNum();//读取行数(从0计数)
        System.out.println(rowNum);
        Row row=sheet.getRow(0);//获得行
        List<String> columns = new ArrayList<String>();
        for (Cell cell : row) {
            String column = cell.getStringCellValue().trim();
            if(!column.equals("") && !column.equals("\n")){
                char[] charArray = column.toCharArray();
                StringBuilder pinyin = new StringBuilder();
                for(int i=0; i<charArray.length; i++){
                    if(Character.toString(charArray[i]).matches("[\\u4E00-\\u9FA5]+")){  //判断是否为中文
                        String value = PinyinHelper.toHanyuPinyinStringArray(charArray[i])[0];
                        value = value.substring(0,value.length()-1);
                        if(value.contains(":")){
                            value = value.replace("u:","v");
                        }
                        pinyin.append(value);
                    }
                }
                columns.add(pinyin.toString());
            }
        }
        // 创建表
        StringBuilder createSQL = new StringBuilder("create table t_hotel(ID VARCHAR PRIMARY KEY");
        StringBuilder insertSQL = new StringBuilder("upsert into t_hotel values(?");
        columns.remove(0); //移除第一条   属于硬编码，
        for (String column : columns) {
            column = ", " + column + " varchar";
            createSQL.append(column);
            insertSQL.append(",?");
        }
        createSQL.append(")");
        insertSQL.append(")");
        System.out.println(createSQL);
        System.out.println(insertSQL);
        // 建表
        createTable(createSQL.toString());
        // 读取Excel文件内容
        // 批量插入数据
        Connection connection = getConnection();
        connection.setAutoCommit(false);
        try {
            PreparedStatement preparedStatement = connection.prepareStatement(insertSQL.toString());
            // 批量插入数据
            for (Row cells : sheet) {
                if (cells.getRowNum() == 0){
                    continue;//跳过第一行
                }
                // 获取最后一个单元格位置
                int lastCellNum = cells.getLastCellNum();

                for(int i = 1; i <= lastCellNum;i++){
                    Cell cell = cells.getCell(i,Row.MissingCellPolicy.RETURN_BLANK_AS_NULL);
                    if(cell!= null){
                        preparedStatement.setString(i,cell.getStringCellValue());
                    }else {
                        preparedStatement.setString(i,""); //设置空字符串
                    }
                }
                preparedStatement.addBatch();
                if (cells.getRowNum()%1000 == 0){
                    preparedStatement.executeBatch();
                }
            }
            // 批量操作
            preparedStatement.executeBatch();
            connection.commit();
            connection.close();
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }
}
